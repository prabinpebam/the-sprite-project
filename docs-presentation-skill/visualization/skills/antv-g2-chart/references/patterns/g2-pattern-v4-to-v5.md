## Change 1: Import Style

```javascript
// G2 v4 syntax
import G2 from '@antv/g2';
const chart = new G2.Chart({ container: 'container' });

// G2 v5 syntax
import { Chart } from '@antv/g2';
const chart = new Chart({ container: 'container' });
```

## Change 2: Data Binding

```javascript
// G2 v4: call .data() separately
chart.data(data);

// G2 v5 (approach 1): pass it on the mark
chart.options({ type: 'interval',  data, encode: {...} });

// G2 v5 (approach 2): data field in options
chart.options({
  type: 'interval',
   data,    // data as a field in options
  encode: { x: 'month', y: 'value' },
});
```

## Change 3: Field Encoding API

```javascript
// G2 v4: chained .encode() methods
chart.interval()
  .encode('x', 'month')
  .encode('y', 'value')
  .encode('color', 'product');

// G2 v5: encode as an object field
chart.options({
  type: 'interval',
  encode: {
    x: 'month',
    y: 'value',
    color: 'product',
  },
});
```

## Change 4: Data Transform (transform)

```javascript
// G2 v4: different method names and parameters
chart.interval().adjust('stack');     // Stack
chart.interval().adjust('dodge');     // Group

// G2 v5: use the transform array
chart.options({
  type: 'interval',
  transform: [{ type: 'stackY' }],   // Stack
  // transform: [{ type: 'dodgeX' }],  // Group
});
```

## Change 5: Coordinate System Configuration

```javascript
// G2 v4
chart.coordinate('polar');
chart.coordinate('theta');
chart.coordinate().transpose();

// G2 v5: configure through the coordinate field object
chart.options({
  coordinate: { type: 'polar' },
  // coordinate: { type: 'theta' },
  // coordinate: { transform: [{ type: 'transpose' }] },
});
```

## Change 6: Helper Annotations (guide to annotation)

```javascript
// G2 v4: guide API
chart.guide().line({ start: ['min', 50], end: ['max', 50] });
chart.guide().text({ position: ['median', 'median'], content: 'Median Line' });

// G2 v5: use lineY/lineX marks (inside view + children)
chart.options({
  type: 'view',
  data,
  children: [
    { type: 'line', encode: { x: 'month', y: 'value' } },
    {
      type: 'lineY',
       [{ threshold: 50 }],
      encode: { y: 'threshold' },
      style: { stroke: 'red', lineDash: [4, 4] },
      labels: [{ text: 'Target Line', position: 'right' }],
    },
  ],
});
```

## Change 7: Label Configuration

```javascript
// G2 v4: .label() method
chart.interval().label('value');
chart.interval().label({ fields: ['value'], formatter: (v) => `${v}0K` });

// G2 v5: labels array (note the plural form)
chart.options({
  type: 'interval',
  labels: [
    {
      // Recommended: access the full datum directly with a text function
      text: (d) => `${d.value}0K`,
      // Or: text: 'value' with formatter formatting
      // formatter: (val) => `${val}0K`,  // val is the value mapped by text, not the datum
    },
  ],
});
```

## Change 8: Tooltip Configuration

```javascript
// G2 v4
chart.tooltip({
  showTitle: false,
  itemTpl: '<li>{name}: {value}</li>',
});

// G2 v5
chart.options({
  tooltip: {
    title: (d) => d.month,
    items: [
      { field: 'value', name: 'Value', valueFormatter: (v) => `${v}0K` },
    ],
  },
  interaction: [{ type: 'tooltip' }],
});
```

## Change 9: Multi-Mark Overlay

```javascript
// G2 v4: multiple chart method calls overlay directly
chart.line().encode('x', 'month').encode('y', 'value');
chart.point().encode('x', 'month').encode('y', 'value');

// G2 v5: use type: 'view' + children
chart.options({
  type: 'view',
  data,
  encode: { x: 'month', y: 'value' },
  children: [
    { type: 'line' },
    { type: 'point' },
  ],
});
```

## Change 10: Animation Configuration

```javascript
// G2 v4
chart.animate(true);

// G2 v5
chart.options({
  animate: {
    enter: { type: 'fadeIn', duration: 300 },
    update: { type: 'morphing', duration: 500 },
  },
});
```

## Quick Check for v4 Error Patterns

When reviewing code generated by an LLM, check for the following common v4 error patterns:

- [ ] `import G2 from '@antv/g2'` (should be a named import)
- [ ] `chart.source(data)` or `chart.data(data)` (should be inside options)
- [ ] `.position('x*y')` (v4 syntax)
- [ ] `.adjust('stack')` / `.adjust('dodge')` (should be a transform array)
- [ ] `chart.guide().line()` (should be a lineY/lineX mark)
- [ ] `.label()` method (should be the `labels` array)
- [ ] Multiple `chart.options()` calls (should use view + children)
